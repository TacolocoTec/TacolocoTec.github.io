                import java.util.List;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // Leer los valores de N y M
        int N = scanner.nextInt();
        int M = scanner.nextInt();

        // Leer la primera lista
        List listA = new ArrayList<>();
        for (int i = 0; i < N; i++) {
            listA.add(scanner.nextInt());
        }

        // Leer la segunda lista
        List listB = new ArrayList<>();
        for (int i = 0; i < M; i++) {
            listB.add(scanner.nextInt());
        }

        // Fusionar las dos listas
        List mergedList = mergeSortedLists(listA, listB);

        // Imprimir la lista fusionada
        for (int number : mergedList) {
            System.out.print(number + " ");
        }
    }

    // Método para fusionar dos listas ordenadas
    private static List mergeSortedLists(List listA, List listB) {
        List mergedList = new ArrayList<>();
        int i = 0, j = 0;

        // Fusionar las listas
        while (i < listA.size() && j < listB.size()) {
            if (listA.get(i) < listB.get(j)) {
                mergedList.add(listA.get(i));
                i++;
            } else {
                mergedList.add(listB.get(j));
                j++;
            }
        }

        // Agregar elementos restantes de listA
        while (i < listA.size()) {
            mergedList.add(listA.get(i));
            i++;
        }

        // Agregar elementos restantes de listB
        while (j < listB.size()) {
            mergedList.add(listB.get(j));
            j++;
        }

        return mergedList;
    }
}
