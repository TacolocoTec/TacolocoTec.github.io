                import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // Leer el número de elementos N
        int n = scanner.nextInt();
        int[] numbers = new int[n];

        // Leer los N números
        for (int i = 0; i < n; i++) {
            numbers[i] = scanner.nextInt();
        }

        // Leer el número K
        int k = scanner.nextInt();

        // Calcular la posición ordinal de K
        int position = findPosition(numbers, k);

        // Imprimir la posición
        System.out.println(position);
    }

    public static int findPosition(int[] numbers, int k) {
        // Ordenar el arreglo
        Arrays.sort(numbers);

        // Encontrar la posición del número K
        for (int i = 0; i < numbers.length; i++) {
            if (numbers[i] == k) {
                return i;
            }
        }

        // En caso de que no se encuentre (esto no debería suceder según el problema)
        return -1;
    }
}
