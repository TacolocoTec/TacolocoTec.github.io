import java.util.*;
                import java.io.*;
                
                public class Main {
                    public static void main(String[] args) throws IOException {
                        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
                        StringBuilder sb = new StringBuilder();
                
                        // Leer los tamaños de las listas
                        String[] tamaños = br.readLine().split(" ");
                        int N = Integer.parseInt(tamaños[0]);
                        int M = Integer.parseInt(tamaños[1]);
                
                        // Leer la lista del gobierno
                        String[] listaGobiernoStr = br.readLine().split(" ");
                        Set listaGobierno = new HashSet<>(N);
                        for (int i = 0; i < N; i++) {
                            listaGobierno.add(Integer.parseInt(listaGobiernoStr[i]));
                        }
                
                        // Leer la lista de Susana Distancia
                        String[] listaSusanaStr = br.readLine().split(" ");
                        Set listaSusana = new HashSet<>(M);
                        for (int i = 0; i < M; i++) {
                            listaSusana.add(Integer.parseInt(listaSusanaStr[i]));
                        }
                
                        // Determinar los números que faltan
                        List faltantes = new ArrayList<>();
                        for (int numero : listaGobierno) {
                            if (!listaSusana.contains(numero)) {
                                faltantes.add(numero);
                            }
                        }
                
                        // Ordenar los números faltantes
                        Collections.sort(faltantes);
                
                        // Construir la salida
                        sb.append(faltantes.size()).append("\n");
                        for (int i = 0; i < faltantes.size(); i++) {
                            if (i > 0) sb.append(" ");
                            sb.append(faltantes.get(i));
                        }
                        if (!faltantes.isEmpty()) sb.append("\n");
                
                        // Imprimir el resultado
                        System.out.print(sb);
                    }
                }
